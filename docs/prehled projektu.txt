function gcall(fn){
  var args=[].slice.call(arguments,1);
  return new Promise(function(res,rej){
    var r=google.script.run.withSuccessHandler(res).withFailureHandler(rej);
    r[fn].apply(r,args);
  });
}
function errMsg(e){ return (e && e.message) ? e.message : (""+e); }
function setSaving(delta){
  savingCount=Math.max(0,savingCount+delta);
  var el=document.getElementById("fm-saving"); if(el) el.style.display=savingCount>0?"inline":"none";
}
function ensureShape(o){
  o=o||{};
  o.projectOrder=o.projectOrder||[]; o.columnOrder=o.columnOrder||[];
  o.manualProjects=o.manualProjects||[]; o.manualColumns=o.manualColumns||[];
  o.manualCells=o.manualCells||{}; o.projectDeadlines=o.projectDeadlines||{};
  o.deletedProjects=o.deletedProjects||[]; o.deletedColumns=o.deletedColumns||[];
  o.notes=o.notes||{}; o.hiddenPhases=o.hiddenPhases||[];
  return o;
}
function rm(a,v){ var i=a.indexOf(v); if(i>=0) a.splice(i,1); }

function applyOp(o, op, d){
  ensureShape(o);
  if(op==="addProject"){
    var i=-1; for(var k=0;k<o.manualProjects.length;k++) if(o.manualProjects[k].name===d.name) i=k;
    if(i>=0) o.manualProjects[i]={name:d.name,deadline:d.deadline}; else o.manualProjects.push({name:d.name,deadline:d.deadline});
    rm(o.deletedProjects,d.name);
  } else if(op==="addColumn"){
    var ex=false; for(var j=0;j<o.manualColumns.length;j++) if(o.manualColumns[j].label===d.label) ex=true;
    if(!ex) o.manualColumns.push({label:d.label,todo:d.todo,task:d.task});
    rm(o.deletedColumns,d.label);
  } else if(op==="setCell"){ o.manualCells[d.key]=d.value; }
  else if(op==="clearCell"){ delete o.manualCells[d.key]; }
  else if(op==="setProjectDeadline"){ if(d.value) o.projectDeadlines[d.name]=d.value; else delete o.projectDeadlines[d.name]; }
  else if(op==="deleteProject"){ if(o.deletedProjects.indexOf(d.name)<0) o.deletedProjects.push(d.name); }
  else if(op==="deleteColumn"){ if(o.deletedColumns.indexOf(d.label)<0) o.deletedColumns.push(d.label); }
  else if(op==="showColumn"){ rm(o.deletedColumns,d.label); }
  else if(op==="hidePhase"){ if(o.hiddenPhases.indexOf(d.todo)<0) o.hiddenPhases.push(d.todo); }
  else if(op==="showPhase"){ rm(o.hiddenPhases,d.todo); }
  else if(op==="showAllCols"){ o.deletedColumns=[]; o.hiddenPhases=[]; }
  else if(op==="hideAllCols"){ o.hiddenPhases=(d.phases||[]).slice(); o.deletedColumns=(d.labels||[]).slice(); }
  else if(op==="setNote"){ if(d.text) o.notes[d.key]=d.text; else delete o.notes[d.key]; }
  else if(op==="setOrder"){ if(d.projectOrder) o.projectOrder=d.projectOrder; if(d.columnOrder) o.columnOrder=d.columnOrder; }
  else if(op==="reset"){ o.projectOrder=[];o.columnOrder=[];o.manualProjects=[];o.manualColumns=[];o.manualCells={};o.projectDeadlines={};o.deletedProjects=[];o.deletedColumns=[];o.notes={};o.hiddenPhases=[]; }
  return o;
}
function pushPatch(op, data){
  applyOp(overlay, op, data); draw();
  setSaving(1);
  return gcall("applyPatch", JSON.stringify({op:op,data:data}))
    .then(function(srv){ overlay=ensureShape(srv); draw(); })
    .catch(function(e){ showError("Uložení se nepovedlo: "+errMsg(e)+" – zkus stránku obnovit."); })
    .then(function(){ setSaving(-1); });
}

function isDone(s){ return String(s||"").trim().toLowerCase().indexOf("hotov")===0; }
function normalizePayload(payload){
  var arr = payload;
  if(arr && !Array.isArray(arr) && typeof arr==="object"){
    var keys=["data","values","rows","result","tasks","items","records"];
    for(var i=0;i<keys.length;i++){ if(Array.isArray(arr[keys[i]])){ arr=arr[keys[i]]; break; } }
  }
  if(!Array.isArray(arr) || arr.length===0) return [];
  if(!Array.isArray(arr[0]) && typeof arr[0]==="object"){
    return arr.map(function(o){ return {
      projektFull:o.Projekt||o.projekt||"", ukol:o["Úkol"]||o.Ukol||o.ukol||"",
      odpovedny:o["Odpovědný"]||o.Odpovedny||o.odpovedny||"", stav:o.Stav||o.stav||"",
      termin:o["Termín"]||o.Termin||o.termin||"", odkaz:o.Odkaz||o.odkaz||"",
      projektOdkaz:o.ProjektOdkaz||o.projektOdkaz||""
    };});
  }
  var rows=arr;
  if(Array.isArray(arr[0]) && String(arr[0][0]).trim().toLowerCase()==="projekt") rows=arr.slice(1);
  return rows.map(function(r){ return {
    projektFull:r[0]||"", ukol:r[2]||"", odpovedny:r[3]||"", stav:r[4]||"",
    termin:r[5]||"", odkaz:r[6]||"", projektOdkaz:r[7]||""
  };});
}
function splitProjekt(full){
  var s=String(full);
  var parts = SEP.test(s) ? s.split(SEP) : s.split(SEP_ALT);
  return {proj:(parts.shift()||"").trim(), todo:parts.join(" › ").trim()};
}
function buildApi(rows){
  var projects=new Map(), columns=new Map(), cells={}, seen={};
  rows.forEach(function(r){
    if(r.odkaz){ if(seen[r.odkaz]) return; seen[r.odkaz]=1; }
    var sp=splitProjekt(r.projektFull);
    if(!sp.proj) return;
    var label = sp.todo ? (sp.todo+" - "+r.ukol) : String(r.ukol);
    if(!projects.has(sp.proj)) projects.set(sp.proj,{name:sp.proj,url:r.projektOdkaz||"",todos:new Map(),deadlines:[]});
    var p=projects.get(sp.proj);
    if(!p.url && r.projektOdkaz) p.url=r.projektOdkaz;
    if(sp.todo){
      if(!p.todos.has(sp.todo)) p.todos.set(sp.todo,{total:0,done:0});
      var t=p.todos.get(sp.todo); t.total++; if(isDone(r.stav)) t.done++;
    }
    if(r.termin) p.deadlines.push(r.termin);
    if(!columns.has(label)) columns.set(label,{label:label,todo:sp.todo,task:r.ukol});
    cells[sp.proj+"||"+label] = {status:isDone(r.stav)?"done":"todo", deadline:r.termin||"", owner:r.odpovedny||"", url:r.odkaz||""};
  });
  return {projects:projects,columns:columns,cells:cells};
}

/* ============================ Sloučení + řazení + skupiny ============================ */
function computeRender(){
  var delP={}; overlay.deletedProjects.forEach(function(n){delP[n]=1;});
  var delC={}; overlay.deletedColumns.forEach(function(l){delC[l]=1;});
  var hidP={}; overlay.hiddenPhases.forEach(function(t){hidP[t]=1;});

  var projNames=[];
  api.projects.forEach(function(_,n){ if(!delP[n]) projNames.push(n); });
  overlay.manualProjects.forEach(function(mp){ if(!delP[mp.name] && projNames.indexOf(mp.name)<0) projNames.push(mp.name); });
  projNames = orderBy(projNames, overlay.projectOrder);

  var colLabels=[];
  api.columns.forEach(function(c,l){ if(delC[l]) return; if(c.todo && hidP[c.todo]) return; colLabels.push(l); });
  overlay.manualColumns.forEach(function(mc){ if(delC[mc.label]) return; if(mc.todo && hidP[mc.todo]) return; if(colLabels.indexOf(mc.label)<0) colLabels.push(mc.label); });
  colLabels = orderBy(colLabels, overlay.columnOrder);

  render.projects = projNames.map(function(n){
    var ap=api.projects.get(n);
    return {name:n, manual:!ap, url:ap?ap.url:"", donePhases:0, totalPhases:0, stripLevel:"green", deadline:projectDeadline(n,ap)};
  });

  render.columns = colLabels.map(function(l){
    var c = api.columns.get(l) || findManualCol(l) || {label:l,todo:"",task:l};
    return {label:l, todo:c.todo, task:c.task, manual:!api.columns.has(l)};
  });

  var gmap={}, gorder=[];
  render.columns.forEach(function(col){
    var key=col.todo||"";
    if(!gmap[key]){ gmap[key]={todo:key, columns:[]}; gorder.push(key); }
    gmap[key].columns.push(col);
  });
  render.groups = gorder.map(function(k){ return {todo:k, collapsed:(k!=="" && !!collapsed[k]), columns:gmap[k].columns}; });

  // Hotové fáze i pro ruční fáze/úkoly + nejhorší úroveň úkolu (proužek u projektu).
  // Počítá se z reálných buněk (respektuje ruční přepisy) napříč viditelnými sloupci.
  var sev={green:1, yellow:2, orange:3, none:3, red:4};
  render.projects.forEach(function(p){
    var total=0, done=0, worst=0;
    render.groups.forEach(function(g){
      var t=0, d=0;
      g.columns.forEach(function(col){
        var cf=cellFor(p.name, col.label);
        if(cf.data && cf.data.status){
          t++;
          if(cf.data.status==="done") d++;
          else { var lvl=cf.data.deadline?deadlineLevel(cf.data.deadline):"none"; var r=sev[lvl]||3; if(r>worst) worst=r; }
        }
      });
      if(t>0){ total++; if(d===t) done++; }
    });
    p.totalPhases=total; p.donePhases=done;
    p.stripLevel = worst>=4?"red":(worst===3?"orange":(worst===2?"yellow":"green"));
  });
}
function findManualCol(l){ for(var i=0;i<overlay.manualColumns.length;i++) if(overlay.manualColumns[i].label===l) return overlay.manualColumns[i]; return null; }
function orderBy(items, order){
  if(!order || !order.length) return items.slice();
  var pos={}; order.forEach(function(k,i){pos[k]=i;});
  var known=items.filter(function(k){return pos[k]!=null;}).sort(function(a,b){return pos[a]-pos[b];});
  var unknown=items.filter(function(k){return pos[k]==null;});
  return known.concat(unknown);
}
function projectDeadline(name, ap){
  if(overlay.projectDeadlines[name]) return {value:overlay.projectDeadlines[name], manual:true};
  var ds = ap ? ap.deadlines.slice() : [];
  var mp = overlay.manualProjects.filter(function(m){return m.name===name;})[0];
  if(mp && mp.deadline) ds.push(mp.deadline);
  if(!ds.length) return {value:"", manual:false};
  ds.sort(); return {value:ds[ds.length-1], manual:false};
}
function cellFor(proj,label){
  var k=proj+"||"+label;
  if(overlay.manualCells[k]) return {data:overlay.manualCells[k], overlay:true};
  if(api.cells[k]) return {data:api.cells[k], overlay:false};
  return {data:null, overlay:false};
}
function groupSummary(projName, group){
  var total=0, done=0, deadlines=[], any=false;
  group.columns.forEach(function(col){
    var cf=cellFor(projName, col.label);
    if(cf.data && cf.data.status){ any=true; total++; if(cf.data.status==="done") done++; else if(cf.data.deadline) deadlines.push(cf.data.deadline); }
  });
  if(!any) return null;
  var allDone = total>0 && done===total;
  var term="", level="green";
  if(!allDone){ deadlines.sort(); term = deadlines.length? deadlines[deadlines.length-1] : ""; level = term? deadlineLevel(term) : "none"; }
  return {total:total, done:done, allDone:allDone, term:term, level:level};
}

function esc(s){ return String(s==null?"":s).replace(/[&<>"]/g,function(m){return {"&":"&amp;","<":"&lt;",">":"&gt;",'"':"&quot;"}[m];}); }
function fmtDate(s){ var m=/^(\d{4})-(\d{2})-(\d{2})/.exec(String(s)); return m?(m[3]+"."+m[2]+"."+m[1]):String(s); }
function parseDate(s){ var m=/^(\d{4})-(\d{2})-(\d{2})/.exec(String(s)); return m?new Date(+m[1],+m[2]-1,+m[3]):null; }
function deadlineLevel(dateStr){
  var d=parseDate(dateStr); if(!d) return "none";
  var today=new Date(); today.setHours(0,0,0,0);
  var diff=Math.round((d-today)/86400000);
  if(diff>3) return "green"; if(diff>=0) return "yellow"; if(diff>=-3) return "orange"; return "red";
}

function buildChecklist(){
  var panel=document.getElementById("fm-dd-panel"); panel.innerHTML="";
  var allLabels=[];
  api.columns.forEach(function(_,l){ allLabels.push(l); });
  overlay.manualColumns.forEach(function(mc){ if(allLabels.indexOf(mc.label)<0) allLabels.push(mc.label); });
  allLabels=orderBy(allLabels, overlay.columnOrder);
  var pmap={}, phases=[];
  allLabels.forEach(function(l){
    var c=api.columns.get(l)||findManualCol(l)||{label:l,todo:"",task:l};
    var key=c.todo||"";
    if(!pmap[key]){ pmap[key]={cols:[]}; phases.push(key); }
    pmap[key].cols.push(c);
  });
  if(!phases.length){ panel.innerHTML='<div class="fm-dd-phase"><span class="fm-dd-plain">Žádné sloupce.</span></div>'; return; }
  var hidP={}; overlay.hiddenPhases.forEach(function(t){hidP[t]=1;});
  var hidC={}; overlay.deletedColumns.forEach(function(l){hidC[l]=1;});
  phases.forEach(function(key){
    var g=pmap[key];
    var row=document.createElement("div"); row.className="fm-dd-phase";
    var pVis = key==="" ? true : !hidP[key];
    if(key!==""){
      row.innerHTML='<label><input type="checkbox" data-phase="'+esc(key)+'"'+(pVis?" checked":"")+'> <b>'+esc(key)+'</b></label>';
    } else {
      row.innerHTML='<span class="fm-dd-plain"><b>(bez fáze)</b></span>';
    }
    panel.appendChild(row);
    g.cols.forEach(function(c){
      var vis = pVis && !hidC[c.label];
      var t=document.createElement("div"); t.className="fm-dd-task";
      t.innerHTML='<label><input type="checkbox" data-col="'+esc(c.label)+'"'+(vis?" checked":"")+(key!==""&&!pVis?" disabled":"")+'> <span>'+esc(c.task)+'</span></label>';
      panel.appendChild(t);
    });
  });
}

function sortRows(key){
  var arr=render.projects.slice();
  if(key==="name") arr.sort(function(a,b){return a.name.localeCompare(b.name,"cs");});
  if(key==="term") arr.sort(function(a,b){ return (a.deadline.value||"9999-99-99").localeCompare(b.deadline.value||"9999-99-99"); });
  if(key==="done") arr.sort(function(a,b){
    var ap=a.totalPhases?a.donePhases/a.totalPhases:-1, bp=b.totalPhases?b.donePhases/b.totalPhases:-1; return bp-ap;
  });
  pushPatch("setOrder",{projectOrder:arr.map(function(p){return p.name;})});
}

function exportCsv(){
  var cols=render.columns; var lines=[];
  var head=["Projekt","Termín","Hotové fáze","Poznámka projektu"].concat(cols.map(function(c){return c.label;}));
  lines.push(head.map(csvCell).join(";"));
  render.projects.forEach(function(p){
    var pnote=overlay.notes[noteKeyProject(p.name)]||"";
    var row=[p.name, p.deadline.value?fmtDate(p.deadline.value):"bez termínu", p.donePhases+" / "+p.totalPhases, pnote];
    cols.forEach(function(c){
      var cf=cellFor(p.name,c.label);
      if(!cf.data||!cf.data.status){
        var en=overlay.notes[noteKeyCell(p.name,c.label)];
        row.push(en?"pozn: "+en:"");
      }
      else{
        var st=cf.data.status==="done"?"Hotovo":"Nehotovo";
        var tm=cf.data.deadline?fmtDate(cf.data.deadline):"bez termínu";
        var nt=overlay.notes[noteKeyCell(p.name,c.label)];
        row.push(st+" | "+tm+(cf.data.owner?" | "+cf.data.owner:"")+(nt?" | pozn: "+nt:""));
      }
    });
    lines.push(row.map(csvCell).join(";"));
  });
  // appendix: poznámky fází (nejsou samostatné sloupce)
  var phaseNoteLines=[];
  Object.keys(overlay.notes).forEach(function(k){
    var i=k.indexOf("||§");
    if(i>=0){ phaseNoteLines.push([k.slice(0,i), k.slice(i+3), overlay.notes[k]].map(csvCell).join(";")); }
  });
  if(phaseNoteLines.length){
    lines.push(""); lines.push("Poznámky k fázím:");
    lines.push(["Projekt","Fáze","Poznámka"].map(csvCell).join(";"));
    lines=lines.concat(phaseNoteLines);
  }
  var csv="\uFEFF"+lines.join("\r\n");
  var blob=new Blob([csv],{type:"text/csv;charset=utf-8"});
  var a=document.createElement("a"); a.href=URL.createObjectURL(blob);
  a.download="matice_projektu.csv"; a.click(); URL.revokeObjectURL(a.href);
}
